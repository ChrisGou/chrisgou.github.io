那么我们就开始实现加密吧，我们修改app\controllers\user.js

var User = require('../proxy/user');

var bcrypt = require('bcrypt')

const SALT_WORK_FACTOR = 10

// 新增用户

exports.signup = function (req, res, next) {

    var username = req.body.username

    var password = req.body.password

    var level = req.body.level

    bcrypt.genSalt(SALT_WORK_FACTOR,function(err,salt){

        if(err) return next(err)

        bcrypt.hash(password,salt,function(err,hash){

            if(err) return next(err)

            User.create(username, hash, level, function (err) {

                if (err) {

                    return next(err);

                }

                res.status(200);

                res.send({

                    code: 0,

                    msg: '新增用户成功',

                })

            })

        })

    })

}

// 登录

exports.signin = function (req, res, next) {

    var username = validator.trim(validator.toString(req.body.username)).toLowerCase()

    var password = validator.trim(validator.toString(req.body.password))

    function sendData(code, msg) {

        res.status(500);

        res.send({

            code: code,

            msg: msg,

        })

    }

    if(username === '') {

        sendData(Constants.USERNAME_ERROR, '缺少用户名');

        return;

    }

    if(password === '') {

        sendData(Constants.PASSWORD_ERROR, '缺少密码');

        return;

    }

    User.getByUserName(username, function (err, user) {

        if (err) {

            return next(err);

        }

        if (!user) {

            sendData(Constants.USERNAME_ERROR, '用户名错误');

            return;

        }

        if(user.deleted){

            sendData(Constants.DELETED_ERROR, '该用户已被注销');

            return;

        }

        bcrypt.compare(password, user.password, function (err, bool) { //密码校验

            if (!bool) {

                sendData(Constants.PASSWORD_ERROR, '密码错误');

                return;

            }

 

            res.status(200);

            res.send({

                code: Constants.SUCCESS,

                msg: '登录成功',

                data: _.pick(user, ['username', '_id'])

            })

        })

    })

}

//需要登陆

exports.userRequired = function (req, res, next) {

    if (!req.session || !req.session.user) {

        return res.status(200).send({

            code: Constants.NO_SIGNIN,

            msg: '未登录',

        });

    }

    next();

}
